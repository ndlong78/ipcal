## Requirements

```sh
- Flask>=2.0.2

```
Bước 2: Tải xuống các gói Python vào thư mục packages

```sh
pip download -d ./packages -r requirements.txt

```
Bước 4: Chỉnh sửa file  Dockerfile

```sh

# Sử dụng image Python chính thức
FROM python:3.10-slim

# Đặt biến môi trường để đảm bảo các đầu vào không bị hỏi trong quá trình cài đặt
ENV PYTHONUNBUFFERED 1

# Sao chép thư mục chứa các gói Python vào container
COPY packages /packages

# Tạo thư mục làm việc
WORKDIR /app

# Sao chép tệp yêu cầu vào thư mục làm việc
COPY requirements.txt /app/

# Cài đặt các thư viện Python từ thư mục local
RUN pip install --no-cache-dir --no-index --find-links=/packages -r requirements.txt

# Sao chép toàn bộ mã nguồn vào thư mục làm việc
COPY . /app/

# Chạy ứng dụng
CMD ["python3", "run.py"]
```
